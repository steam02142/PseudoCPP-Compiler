FUNCTION add ( ) -> int
    SET a TO 0
    SET b TO 0
    PRINT "Enter first number:"
    READ a
    PRINT "Enter second number:"
    READ b
    RETURN ( a + b )
ENDFUNCTION

FUNCTION subtract ( ) -> int
    SET a TO 0
    SET b TO 0
    PRINT "Enter first number:"
    READ a
    PRINT "Enter second number:"
    READ b
    RETURN ( a - b )
ENDFUNCTION

FUNCTION multiply ( ) -> int
    SET a TO 0
    SET b TO 0
    PRINT "Enter first number:"
    READ a
    PRINT "Enter second number:"
    READ b
    RETURN ( a * b )
ENDFUNCTION

FUNCTION divide ( ) -> int
    SET a TO 0
    SET b TO 0
    PRINT "Enter numerator:"
    READ a
    PRINT "Enter denominator:"
    READ b
    IF b == 0
        PRINT "Error: Division by zero"
        RETURN 0
    ENDIF
    RETURN ( a / b )
ENDFUNCTION

MAIN
    SET choice TO 0
    SET result TO 0

    PRINT "Math Menu:"
    PRINT "1. Add"
    PRINT "2. Subtract"
    PRINT "3. Multiply"
    PRINT "4. Divide"
    PRINT "Enter your choice:"
    READ choice

    IF choice == 1
        CALL add RETURNING result
    ELSEIF choice == 2
        CALL subtract RETURNING result
    ELSEIF choice == 3
        CALL multiply RETURNING result
    ELSEIF choice == 4
        CALL divide RETURNING result
    ELSE
        PRINT "Invalid choice"
    ENDIF

    PRINT "Result:"
    PRINT result

ENDMAIN

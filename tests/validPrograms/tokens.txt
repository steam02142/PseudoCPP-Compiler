FUNCTION fib ( n: int ) -> int
    IF n <= 1
        RETURN n
    ENDIF

    CALL fib WITH ( n - 1 ) RETURNING a
    CALL fib WITH ( n - 2 ) RETURNING b

    RETURN ( a + b )
ENDFUNCTION
